# a fresh local venv that should run faster-whisper/ctranslate2 on GPU:
#
#   pip install -r services/core-api/requirements-torch-cu126.txt
#   pip install -r services/core-api/requirements.txt
#
# Do not install these packages from the default PyPI index if CUDA support
# is required; use the PyTorch cu126 wheel index below.

#   torch==2.7.1+cu126
#   torchaudio==2.7.1+cu126
#   torchvision==0.22.1+cu126
#   ctranslate2==4.7.1     (in requirements.txt via faster-whisper)
#   faster-whisper==1.2.1  (in requirements.txt)
#
# The torch wheel pulls the correct nvidia-* transitive dependencies
# for cu126 automatically. Do not add nvidia-* packages here manually.
#
